public void quickSort(int[] arr, int low, int high) { //verifica si el inidice inferior es menor al superior, llama a partition(); divide al arreglo en dos partes y se llama a si mismo para ordenar cada subarreglo
        if (low < high) {
            int pivotIndex = partition(arr, low, high);

            quickSort(arr, low, pivotIndex - 1);
            quickSort(arr, pivotIndex + 1, high);
        }

public int sum(int[] nums) {
        int total = 0;
        for (int n : nums) {
            total += n;
        }
        return total;
    }
